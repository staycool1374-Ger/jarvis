49inline void chacha_quarter_round(uint32_t& a, uint32_t& b, uint32_t& c,
50 uint32_t& d) {
51 a += b; d ^= a; d = (d << 16) | (d >> 16);
52 c += d; b ^= c; b = (b << 12) | (b >> 20);
53 a += b; d ^= a; d = (d << 8) | (d >> 24);
54 c += d; b ^= c; b = (b << 7) | (b >> 25);
55}

62inline void chacha_init_state(ChaChaState& state, const uint8_t key[CHACHA_KEY_SIZE],
63 uint32_t counter, const uint8_t nonce[CHACHA_NONCE_SIZE]) {
64 // Constants: "expand 32-byte k"
65 state.s[0] = 0x61707865;
66 state.s[1] = 0x3320646e;
67 state.s[2] = 0x79622d32;
68 state.s[3] = 0x6b206574;
69
70 // Key (8 × uint32_t, little-endian)
71 for (size_t i = 0; i < 8; ++i) {
72 state.s[4 + i] = static_cast<uint32_t>(key[4 * i])
73 | (static_cast<uint32_t>(key[4 * i + 1]) << 8)
74 | (static_cast<uint32_t>(key[4 * i + 2]) << 16)
75 | (static_cast<uint32_t>(key[4 * i + 3]) << 24);
76 }
77
78 // Block counter (uint32_t)
79 state.s[12] = counter;
80
81 // Nonce (3 × uint32_t, little-endian)
82 for (size_t i = 0; i < 3; ++i) {
83 state.s[13 + i] = static_cast<uint32_t>(nonce[4 * i])
84 | (static_cast<uint32_t>(nonce[4 * i + 1]) << 8)
85 | (static_cast<uint32_t>(nonce[4 * i + 2]) << 16)
86 | (static_cast<uint32_t>(nonce[4 * i + 3]) << 24);
87 }
88}
