100inline void validate_all_consistency() {
101 size_t all_count = 0;
102 size_t sum_individual = 0;
103 for (size_t i = 0; i < k_expected_count_size; ++i) {
104 size_t c = arch_count(k_expected_counts[i]);
105 if (__builtin_strcmp(k_expected_counts[i].name, "all") == 0) {
106 all_count = c;
107 } else if (__builtin_strcmp(k_expected_counts[i].name, "safe") != 0) {
108 sum_individual += c;
109 }
110 }
111 if (all_count > 0 && sum_individual < all_count) {
112 Logger::warn("[TCOUNT] CONSISTENCY: sum(individual)=%u < all=%u -- "
113 "some tests missing from individual class entries",
114 (unsigned)sum_individual, (unsigned)all_count);
115 } else if (all_count > 0) {
116 Logger::info(
117 "[TCOUNT] CONSISTENCY: sum(individual)=%u >= all=%u (overlap=%u)",
118 (unsigned)sum_individual, (unsigned)all_count,
119 (unsigned)(sum_individual - all_count));
120 }
