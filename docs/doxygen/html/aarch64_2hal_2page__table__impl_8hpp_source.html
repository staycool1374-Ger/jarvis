144inline uint64_t *ArchPageTable::get_table(uint64_t table_base, uint64_t index,
145 bool create) {
146 uint64_t *table = reinterpret_cast<uint64_t *>(table_base);
147 uint64_t entry = table[index];
148
149 if (entry & DESC_VALID) {
150 if (entry & DESC_TABLE) {
151 return reinterpret_cast<uint64_t *>(entry & ~0xFFF);
152 }
153 return nullptr;
154 }
155
156 if (!create)
157 return nullptr;
158
159 uint64_t new_page = kernel::PMM::alloc_page_table();
160 if (!new_page)
161 return nullptr;
162
163 void *new_page_ptr = reinterpret_cast<void *>(new_page);
164 memset(new_page_ptr, 0, PAGE_SIZE);
165 table[index] = new_page | DESC_VALID | DESC_TABLE;
166 return reinterpret_cast<uint64_t *>(new_page);
167}

169inline uint64_t ArchPageTable::attr_from_flags(uint64_t flags) {
170 uint64_t attr = DESC_VALID | DESC_AF | DESC_SH_INNER | DESC_NS;
171 if (flags & kernel::PageFlags::PRESENT)
172 attr |= DESC_VALID;
173 if (flags & kernel::PageFlags::WRITE)
174 attr |= AP_RW;
175 else
176 attr |= AP_RO;
177 if (flags & kernel::PageFlags::USER) {
178 // User accessible
179 } else {
180 attr |= UXN | PXN;
181 }
182 if (flags & kernel::PageFlags::NX)
183 attr |= UXN | PXN;
184 if (flags & kernel::PageFlags::CACHE_DISABLED)
185 attr |= ATTR_IDX_DEVICE;
186 else
187 attr |= ATTR_IDX_NORMAL;
188 return attr;
189}

211inline kernel::ErrorOr<void>
212ArchPageTable::map_page(uint64_t virt, uint64_t phys, uint64_t flags) {
213 uint64_t l0_idx = (virt >> L0_SHIFT) & TABLE_MASK;
214 uint64_t l1_idx = (virt >> L1_SHIFT) & TABLE_MASK;
215 uint64_t l2_idx = (virt >> L2_SHIFT) & TABLE_MASK;
216 uint64_t l3_idx = (virt >> L3_SHIFT) & TABLE_MASK;
217
218 uint64_t l0_phys = current();
219
220 uint64_t *l1 = walk_l0(l0_phys, l0_idx, true);
221 if (!l1)
222 return kernel::Error::OOM;
223
224 uint64_t *l2 = walk_l1(reinterpret_cast<uint64_t>(l1), l1_idx, true);
225 if (!l2)
226 return kernel::Error::OOM;
227
228 uint64_t entry_flags = attr_from_flags(flags);
229
230 if ((virt & ((1ULL << L2_SHIFT) - 1)) == 0 &&
231 (phys & ((1ULL << L2_SHIFT) - 1)) == 0) {
232 l2[l2_idx] = phys | entry_flags | DESC_BLOCK | (1ULL << 10);
233 dsb_sy();
234 isb();
235 return {};
236 }
237
238 uint64_t *l3 = walk_l2(reinterpret_cast<uint64_t>(l2), l2_idx, true);
239 if (!l3)
240 return kernel::Error::OOM;
241
242 l3[l3_idx] = phys | entry_flags | DESC_BLOCK | (1ULL << 10);
243 dsb_sy();
244 isb();
245 return {};
246}

248inline kernel::ErrorOr<void> ArchPageTable::unmap_page(uint64_t virt) {
249 uint64_t l0_idx = (virt >> L0_SHIFT) & TABLE_MASK;
250 uint64_t l1_idx = (virt >> L1_SHIFT) & TABLE_MASK;
251 uint64_t l2_idx = (virt >> L2_SHIFT) & TABLE_MASK;
252 uint64_t l3_idx = (virt >> L3_SHIFT) & TABLE_MASK;
253
254 uint64_t l0_phys = current();
255 uint64_t *l0 = reinterpret_cast<uint64_t *>(l0_phys);
256
257 if (!(l0[l0_idx] & DESC_VALID))
258 return kernel::Error::NOT_FOUND;
259 uint64_t *l1 = reinterpret_cast<uint64_t *>(l0[l0_idx] & ~0xFFF);
260 if (!(l1[l1_idx] & DESC_VALID))
261 return kernel::Error::NOT_FOUND;
262 uint64_t *l2 = reinterpret_cast<uint64_t *>(l1[l1_idx] & ~0xFFF);
263
264 if ((virt & ((1ULL << L2_SHIFT) - 1)) == 0 && (l2[l2_idx] & DESC_BLOCK)) {
265 l2[l2_idx] = 0;
266 dsb_sy();
267 isb();
268 return {};
269 }
270
271 if (!(l2[l2_idx] & DESC_VALID))
272 return kernel::Error::NOT_FOUND;
273 uint64_t *l3 = reinterpret_cast<uint64_t *>(l2[l2_idx] & ~0xFFF);
274
275 if (!(l3[l3_idx] & DESC_VALID))
276 return kernel::Error::NOT_FOUND;
277 l3[l3_idx] = 0;
278 dsb_sy();
279 isb();
280 return {};
281}

283inline uint64_t ArchPageTable::get_physical(uint64_t virt) {
284 uint64_t l0_idx = (virt >> L0_SHIFT) & TABLE_MASK;
285 uint64_t l1_idx = (virt >> L1_SHIFT) & TABLE_MASK;
286 uint64_t l2_idx = (virt >> L2_SHIFT) & TABLE_MASK;
287 uint64_t l3_idx = (virt >> L3_SHIFT) & TABLE_MASK;
288
289 uint64_t l0_phys = current();
290 uint64_t *l0 = reinterpret_cast<uint64_t *>(l0_phys);
291
292 if (!(l0[l0_idx] & DESC_VALID))
293 return 0;
294 uint64_t *l1 = reinterpret_cast<uint64_t *>(l0[l0_idx] & ~0xFFF);
295 if (!(l1[l1_idx] & DESC_VALID))
296 return 0;
297 uint64_t *l2 = reinterpret_cast<uint64_t *>(l1[l1_idx] & ~0xFFF);
298
299 if ((virt & ((1ULL << L2_SHIFT) - 1)) == 0 && (l2[l2_idx] & DESC_BLOCK)) {
300 return (l2[l2_idx] & ~0x1FFFFF) | (virt & 0x1FFFFF);
301 }
302
303 if (!(l2[l2_idx] & DESC_VALID))
304 return 0;
305 uint64_t *l3 = reinterpret_cast<uint64_t *>(l2[l2_idx] & ~0xFFF);
306 if (!(l3[l3_idx] & DESC_VALID))
307 return 0;
308
309 return (l3[l3_idx] & ~0xFFF) | (virt & 0xFFF);
310}
