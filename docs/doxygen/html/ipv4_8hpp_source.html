33struct Ipv4Addr {
34 uint8_t addr[IPV4_ADDR_LEN];
35
36 bool operator==(const Ipv4Addr& o) const {
37 for (size_t i = 0; i < IPV4_ADDR_LEN; ++i)
38 if (addr[i] != o.addr[i]) return false;
39 return true;
40 }
41
42 uint32_t as_u32() const {
43 return (static_cast<uint32_t>(addr[0]) << 24)
44 | (static_cast<uint32_t>(addr[1]) << 16)
45 | (static_cast<uint32_t>(addr[2]) << 8)
46 | static_cast<uint32_t>(addr[3]);
47 }

49 static Ipv4Addr from_u32(uint32_t v) {
50 Ipv4Addr a;
51 a.addr[0] = static_cast<uint8_t>((v >> 24) & 0xFF);
52 a.addr[1] = static_cast<uint8_t>((v >> 16) & 0xFF);
53 a.addr[2] = static_cast<uint8_t>((v >> 8) & 0xFF);
54 a.addr[3] = static_cast<uint8_t>(v & 0xFF);
55 return a;
56 }
57} __attribute__((packed));

81inline uint16_t ipv4_checksum(const Ipv4Header* hdr) {
82 uint32_t sum = 0;
83 auto* words = reinterpret_cast<const uint16_t*>(hdr);
84 for (int i = 0; i < 10; ++i) { // first 20 bytes = 10 half-words
85 sum += words[i];
86 }
87 while (sum >> 16) sum = (sum & 0xFFFF) + (sum >> 16);
88 return static_cast<uint16_t>(~sum & 0xFFFF);
89}
