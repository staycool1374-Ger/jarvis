144inline void qemu_debug_exit(uint8_t code) {
145 // ARM Semihosting SYS_EXIT (0x18) with ADP_Stopped_ApplicationExit
146 uint32_t params[2] = {0x18, code};
147 uint64_t addr = reinterpret_cast<uint64_t>(params);
148 __asm__ volatile("mov x0, #0x18\n\t"
149 "mov x1, %0\n\t"
150 "hlt #0xf000"
151 :
152 : "r"(addr)
153 : "x0", "x1", "memory");
154 while (1) {
155 asm volatile("wfi");
156 }
157}

242inline void fp_enable() {
243 uint64_t cpacr{};
244 asm volatile("mrs %0, cpacr_el1" : "=r"(cpacr));
245 cpacr |= (3 << 20); // FPEN: trap disabled
246 asm volatile("msr cpacr_el1, %0" : : "r"(cpacr) : "memory");
247 asm volatile("isb");
248}
