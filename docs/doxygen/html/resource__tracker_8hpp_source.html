48 void track_mempool_alloc(size_t pool_idx) {
49 if (pool_idx < MemPool::POOL_COUNT)
50 ++counters_.mempool_used[pool_idx];
51 }
52 void track_mempool_free(size_t pool_idx) {
53 if (pool_idx < MemPool::POOL_COUNT && counters_.mempool_used[pool_idx] > 0)
54 --counters_.mempool_used[pool_idx];
55 }
56 void track_pmm_alloc(size_t pages) { counters_.pmm_pages_used += pages; }
57 void track_pmm_free(size_t pages) {
58 counters_.pmm_pages_used -= (counters_.pmm_pages_used >= pages)
59 ? pages : counters_.pmm_pages_used;
60 }
61 void track_task_add() { ++counters_.tasks; }
