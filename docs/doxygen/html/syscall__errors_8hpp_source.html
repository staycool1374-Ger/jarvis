127// NOLINTNEXTLINE(performance-enum-size)
128enum SyscallError : uint64_t {
129#define X(name, num, msg) SYS_ERR_##name = (num),
130 SYSCALL_ERROR_CODES
131#undef X
132};
133
135template <> inline const char *error_string(SyscallError e) {
136 switch (e) {
137#define X(name, num, msg) \
138 case SYS_ERR_##name: \
139 return msg;
140 SYSCALL_ERROR_CODES
141#undef X
142 }
143 return "Unknown syscall error";
144}
