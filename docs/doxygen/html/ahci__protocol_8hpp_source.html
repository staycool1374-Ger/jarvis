195// ──────────────────────────────────────────────
196// Command List
197// Each entry is a Command Header (32 bytes).
198// 32 entries per port (max).
199// ──────────────────────────────────────────────
200
202constexpr uint32_t AHCI_MAX_CMDS = 32;
203
205struct CmdHeader {
206 // DW0
207 uint16_t cfl; // Command FIS Length in DWORDS (bits 0-4)
208 uint16_t attrs; // Attributes (bits 5-15: A=ATAPI, W=Write, C=ClearBusy, P=Prefetch, R=Reset)
209 // DW1
210 uint16_t prdbc; // Physical Region Descriptor Byte Count (transferred)
211 uint8_t pmp; // Port Multiplier Port
212 uint8_t rsvd0;
213 // DW2
214 uint32_t ctba; // Command Table Descriptor Base Address (low 32)
215 // DW3
216 uint32_t ctbau; // Command Table Descriptor Base Address (upper 32)
217 // DW4–DW7: Reserved
218 uint32_t rsvd1[4];
219} __attribute__((packed));
220
222constexpr uint16_t CMDHDR_CFL_MASK = 0x001F;
223constexpr uint16_t CMDHDR_ATAPI = 0x0020; // Device is ATAPI
224constexpr uint16_t CMDHDR_WRITE = 0x0040; // Write (host→device)
225constexpr uint16_t CMDHDR_CLR_BUSY = 0x0080; // Clear Busy upon R_OK
226constexpr uint16_t CMDHDR_PREFETCH = 0x0100; // Prefetchable
227constexpr uint16_t CMDHDR_RESET = 0x0200; // Reset
228constexpr uint16_t CMDHDR_BIST = 0x0400; // BIST
229constexpr uint16_t CMDHDR_CMP_PMA = 0x0800; // Completion for Port Multiplier

231// ──────────────────────────────────────────────
232// Command Table (128 bytes header + up to 8K PRD entries)
233// ──────────────────────────────────────────────
234
236struct CmdFIS {
237 // DW0
238 uint8_t fis_type; // 0x27
239 uint8_t pm_port_c; // bit 7 = C (is command), bits 0-3 = PM port
240 uint8_t command; // ATA command (READ DMA EXT = 0x25, etc.)
241 uint8_t features;
242 // DW1
243 uint8_t lba_lo;
244 uint8_t lba_mid;
245 uint8_t lba_hi;
246 uint8_t device; // master=0x40, slave=0x50, LBA=0xE0
247 // DW2
248 uint8_t lba_lo_exp;
249 uint8_t lba_mid_exp;
250 uint8_t lba_hi_exp;
251 uint8_t features_exp;
252 // DW3
253 uint8_t count_lo; // sector count (low byte)
254 uint8_t count_hi; // sector count (high byte)
255 uint8_t icc; // Isochronous Command Completion
256 uint8_t control;
257 // DW4
258 uint32_t aux; // auxiliary
259} __attribute__((packed));
260
261static_assert(sizeof(CmdFIS) == 20, "CmdFIS must be 20 bytes");
262
264constexpr uint8_t FIS_TYPE_REG_H2D = 0x27; // Register FIS - Host to Device
265constexpr uint8_t FIS_TYPE_REG_D2H = 0x34; // Register FIS - Device to Host
266constexpr uint8_t FIS_TYPE_DMA_ACT = 0x39; // DMA Activate FIS
267constexpr uint8_t FIS_TYPE_DMA_SETUP = 0x41; // DMA Setup FIS
268constexpr uint8_t FIS_TYPE_DATA = 0x46; // Data FIS
269constexpr uint8_t FIS_TYPE_BIST = 0x58; // BIST Activate FIS
270constexpr uint8_t FIS_TYPE_PIO_SETUP = 0x5F; // PIO Setup FIS
271constexpr uint8_t FIS_TYPE_DEV_BITS = 0xA1; // Set Device Bits FIS

274struct PrdHbaEntry {
275 uint32_t dba; // Data Base Address (low 32)
276 uint32_t dbau; // Data Base Address (upper 32, if CAP & CAP_64BIT)

279} __attribute__((packed));

292struct CmdTable {
293 CmdFIS cfis; // 20 bytes
294 uint8_t atapi_cmd[16];// ATAPI command (if ATAPI)
295 uint8_t rsvd0[92]; // padding to 128 bytes
296 PrdHbaEntry prd[AHCI_MAX_PRD]; // PRD entries
297} __attribute__((packed));
298
299static_assert(sizeof(CmdTable) == 128 + 256 * sizeof(PrdHbaEntry),
300 "CmdTable size mismatch");

302// ──────────────────────────────────────────────
303// Received FIS Structure (from PORT_FB)
304// Aligned to 256 bytes, holds shadow registers + D2H + DMA + PIO + SDB
305// ──────────────────────────────────────────────
306
308struct __attribute__((packed)) ReceivedFis {
309 uint8_t rsvd0[0x20]; // Shadow register buffer (0x00–0x1F)
310 uint32_t d2h_fis[2]; // D2H Register FIS (offset 0x20)
311 uint32_t dma_setup[2]; // DMA Setup FIS (offset 0x28)
312 uint32_t pio_setup[2]; // PIO Setup FIS (offset 0x30)
313 uint32_t sdb_fis[2]; // Set Device Bits FIS (offset 0x38)
314 uint8_t rsvd1[0xA0]; // (offset 0x40–0xDF)
315 uint16_t vendor[0x10]; // Vendor-specific (0xE0–0xFF)
316};
317
318static_assert(sizeof(ReceivedFis) == 0x100, "ReceivedFis must be 256 bytes");
