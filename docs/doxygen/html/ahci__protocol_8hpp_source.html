199// ──────────────────────────────────────────────
200// Command List
201// Each entry is a Command Header (32 bytes).
202// 32 entries per port (max).
203// ──────────────────────────────────────────────
204
206constexpr uint32_t AHCI_MAX_CMDS = 32;

209struct CmdHeader {
210 // DW0
211 uint16_t cfl; // Command FIS Length in DWORDS (bits 0-4)
212 uint16_t attrs; // Attributes (bits 5-15: A=ATAPI, W=Write, C=ClearBusy,
213 // P=Prefetch, R=Reset)
214 // DW1
215 uint16_t prdbc; // Physical Region Descriptor Byte Count (transferred)
216 uint8_t pmp; // Port Multiplier Port
217 uint8_t rsvd0;
218 // DW2
219 uint32_t ctba; // Command Table Descriptor Base Address (low 32)
220 // DW3
221 uint32_t ctbau; // Command Table Descriptor Base Address (upper 32)
222 // DW4–DW7: Reserved
223 uint32_t rsvd1[4];
224} __attribute__((packed));
225
227constexpr uint16_t CMDHDR_CFL_MASK = 0x001F;
228constexpr uint16_t CMDHDR_ATAPI = 0x0020; // Device is ATAPI
229constexpr uint16_t CMDHDR_WRITE = 0x0040; // Write (host→device)
230constexpr uint16_t CMDHDR_CLR_BUSY = 0x0080; // Clear Busy upon R_OK
231constexpr uint16_t CMDHDR_PREFETCH = 0x0100; // Prefetchable
232constexpr uint16_t CMDHDR_RESET = 0x0200; // Reset
233constexpr uint16_t CMDHDR_BIST = 0x0400; // BIST
234constexpr uint16_t CMDHDR_CMP_PMA = 0x0800; // Completion for Port Multiplier

236// ──────────────────────────────────────────────
237// Command Table (128 bytes header + up to 8K PRD entries)
238// ──────────────────────────────────────────────
239
241struct CmdFIS {
242 // DW0
243 uint8_t fis_type; // 0x27
244 uint8_t pm_port_c; // bit 7 = C (is command), bits 0-3 = PM port
245 uint8_t command; // ATA command (READ DMA EXT = 0x25, etc.)
246 uint8_t features;
247 // DW1
248 uint8_t lba_lo;
249 uint8_t lba_mid;
250 uint8_t lba_hi;
251 uint8_t device; // master=0x40, slave=0x50, LBA=0xE0
252 // DW2
253 uint8_t lba_lo_exp;
254 uint8_t lba_mid_exp;
255 uint8_t lba_hi_exp;
256 uint8_t features_exp;
257 // DW3
258 uint8_t count_lo; // sector count (low byte)
259 uint8_t count_hi; // sector count (high byte)
260 uint8_t icc; // Isochronous Command Completion
261 uint8_t control;
262 // DW4
263 uint32_t aux; // auxiliary
264} __attribute__((packed));
265
266static_assert(sizeof(CmdFIS) == 20, "CmdFIS must be 20 bytes");

298struct CmdTable {
299 CmdFIS cfis; // 20 bytes
300 uint8_t atapi_cmd[16]; // ATAPI command (if ATAPI)
301 uint8_t rsvd0[92]; // padding to 128 bytes
302 PrdHbaEntry prd[AHCI_MAX_PRD]; // PRD entries
303} __attribute__((packed));
304
305static_assert(sizeof(CmdTable) == 128 + 256 * sizeof(PrdHbaEntry),
306 "CmdTable size mismatch");

308// ──────────────────────────────────────────────
309// Received FIS Structure (from PORT_FB)
310// Aligned to 256 bytes, holds shadow registers + D2H + DMA + PIO + SDB
311// ──────────────────────────────────────────────
312
314struct __attribute__((packed)) ReceivedFis {
315 uint8_t rsvd0[0x20]; // Shadow register buffer (0x00–0x1F)
316 uint32_t d2h_fis[2]; // D2H Register FIS (offset 0x20)
317 uint32_t dma_setup[2]; // DMA Setup FIS (offset 0x28)
318 uint32_t pio_setup[2]; // PIO Setup FIS (offset 0x30)
319 uint32_t sdb_fis[2]; // Set Device Bits FIS (offset 0x38)
320 uint8_t rsvd1[0xA0]; // (offset 0x40–0xDF)
321 uint16_t vendor[0x10]; // Vendor-specific (0xE0–0xFF)
322};
323
324static_assert(sizeof(ReceivedFis) == 0x100, "ReceivedFis must be 256 bytes");
