135#if defined(CONFIG_ARCH_AARCH64)
136 // aarch64 stage-1 page-table descriptor bit layout.
137 // Table descriptors (L0-L2) and page descriptors (L3): bits[1:0]=11.
138 // Block descriptors (L1-L2): bits[1:0]=01.
139 // AP[1:0] in bits[7:6]; 00=EL1-RW/EL0-*, 01=EL1-RW/EL0-RW,
140 // 10=EL1-RO/EL0-*, 11=EL1-RO/EL0-RO.
141 // AF (Access Flag) must be 1 for stage-1 - bit 8 (NOT bit 10).
142 static constexpr uint64_t PAGE_PRESENT = 1ULL << 0;
143 static constexpr uint64_t PAGE_TABLE = 1ULL << 1;
144 static constexpr uint64_t PAGE_AF = 1ULL << 8; // Access Flag (stage-1)
145 static constexpr uint64_t PAGE_AP_USER = 1ULL << 6; // AP[0] - EL0 accessible
146 static constexpr uint64_t PAGE_AP_RO = 1ULL << 7; // AP[1] - read-only (EL1)
147
148 // Compatibility aliases (used in flag-building expressions)
149 static constexpr uint64_t PAGE_WRITE = 0; // no-op - writable by default
150 static constexpr uint64_t PAGE_USER = PAGE_AP_USER;
151 static constexpr uint64_t PAGE_HUGE = 0; // not a flag; detected via level
152#elif defined(CONFIG_ARCH_RISCV64)
153 // RISC-V Sv39 PTE format (from privilege spec)
154 // bit 0: V (Valid)
155 // bit 1: R (Readable)
156 // bit 2: W (Writable)
157 // bit 3: X (Executable)
158 // bit 4: U (User)
159 // bit 5: G (Global)
160 // bit 6: A (Accessed)
161 // bit 7: D (Dirty)
162 // bit 8: RSW (reserved for software)
163 // bits 9-53: PPN[0..44] (physical page number)
164 // bits 54-63: reserved for future use
165 static constexpr uint64_t PAGE_PRESENT = 1ULL << 0; // V
166 static constexpr uint64_t PAGE_READ = 1ULL << 1; // R
167 static constexpr uint64_t PAGE_WRITE = 1ULL << 2; // W
168 static constexpr uint64_t PAGE_EXEC = 1ULL << 3; // X
169 static constexpr uint64_t PAGE_USER = 1ULL << 4; // U
170 static constexpr uint64_t PAGE_GLOBAL = 1ULL << 5; // G
171 static constexpr uint64_t PAGE_ACCESSED = 1ULL << 6; // A
172 static constexpr uint64_t PAGE_DIRTY = 1ULL << 7; // D
173 // Table entry: V=1, R=W=X=0 (points to next level)
174 static constexpr uint64_t PAGE_TABLE = PAGE_PRESENT; // V=1, no RWX = table pointer
175 // Block/leaf detection: if V=1 and (R|W|X)=1 then leaf page
176 static constexpr uint64_t PAGE_HUGE = 0; // detected via R|W|X bits
177#else
178 static constexpr uint64_t PAGE_PRESENT = 1ULL << 0;
179 static constexpr uint64_t PAGE_WRITE = 1ULL << 1;
180 static constexpr uint64_t PAGE_USER = 1ULL << 2;
181 static constexpr uint64_t PAGE_HUGE = 1ULL << 7;
182#endif
