154#if defined(CONFIG_ARCH_AARCH64)
155 // aarch64 stage-1 page-table descriptor bit layout.
156 // Table descriptors (L0-L2) and page descriptors (L3): bits[1:0]=11.
157 // Block descriptors (L1-L2): bits[1:0]=01.
158 // AP[1:0] in bits[7:6]; 00=EL1-RW/EL0-*, 01=EL1-RW/EL0-RW,
159 // 10=EL1-RO/EL0-*, 11=EL1-RO/EL0-RO.
160 // AF (Access Flag) must be 1 for stage-1 - bit 8 (NOT bit 10).
161 static constexpr uint64_t PAGE_PRESENT = 1ULL << 0;
162 static constexpr uint64_t PAGE_TABLE = 1ULL << 1;
163 static constexpr uint64_t PAGE_AF = 1ULL << 8; // Access Flag (stage-1)
164 static constexpr uint64_t PAGE_AP_USER = 1ULL
165 << 6; // AP[0] - EL0 accessible
166 static constexpr uint64_t PAGE_AP_RO = 1ULL << 7; // AP[1] - read-only (EL1)
167
168 // Compatibility aliases (used in flag-building expressions)
169 static constexpr uint64_t PAGE_WRITE = 0; // no-op - writable by default
170 static constexpr uint64_t PAGE_USER = PAGE_AP_USER;
171 static constexpr uint64_t PAGE_HUGE = 0; // not a flag; detected via level
172#elif defined(CONFIG_ARCH_RISCV64)
173 // RISC-V Sv39 PTE format (from privilege spec)
174 // bit 0: V (Valid)
175 // bit 1: R (Readable)
176 // bit 2: W (Writable)
177 // bit 3: X (Executable)
178 // bit 4: U (User)
179 // bit 5: G (Global)
180 // bit 6: A (Accessed)
181 // bit 7: D (Dirty)
182 // bit 8: RSW (reserved for software)
183 // bits 9-53: PPN[0..44] (physical page number)
184 // bits 54-63: reserved for future use
185 static constexpr uint64_t PAGE_PRESENT = 1ULL << 0; // V
186 static constexpr uint64_t PAGE_READ = 1ULL << 1; // R
187 static constexpr uint64_t PAGE_WRITE = 1ULL << 2; // W
188 static constexpr uint64_t PAGE_EXEC = 1ULL << 3; // X
189 static constexpr uint64_t PAGE_USER = 1ULL << 4; // U
190 static constexpr uint64_t PAGE_GLOBAL = 1ULL << 5; // G
191 static constexpr uint64_t PAGE_ACCESSED = 1ULL << 6; // A
192 static constexpr uint64_t PAGE_DIRTY = 1ULL << 7; // D
193 // Table entry: V=1, R=W=X=0 (points to next level)
194 static constexpr uint64_t PAGE_TABLE =
195 PAGE_PRESENT; // V=1, no RWX = table pointer
196 // Block/leaf detection: if V=1 and (R|W|X)=1 then leaf page
197 static constexpr uint64_t PAGE_HUGE = 0; // detected via R|W|X bits
198#else
199 static constexpr uint64_t PAGE_PRESENT = 1ULL << 0;
200 static constexpr uint64_t PAGE_WRITE = 1ULL << 1;
201 static constexpr uint64_t PAGE_USER = 1ULL << 2;
202 static constexpr uint64_t PAGE_HUGE = 1ULL << 7;
203#endif
