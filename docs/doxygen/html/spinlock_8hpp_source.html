43 void lock() noexcept {
44 while (atomic_exchange(&locked_, 1, __ATOMIC_ACQUIRE)) {
45 arch::pause();
46 }
47 __atomic_store_n(&holder_, __builtin_return_address(0),
48 __ATOMIC_RELAXED);
49 }
50
52 void unlock() noexcept {
53 atomic_store(&locked_, 0, __ATOMIC_RELEASE);
54 __atomic_store_n(&holder_, nullptr, __ATOMIC_RELAXED);
55 }
56
59 void reset() noexcept {
60 locked_ = 0;
61 holder_ = nullptr;
62 }
63
66 bool try_lock() noexcept {
67 bool acquired = !atomic_exchange(&locked_, 1, __ATOMIC_ACQUIRE);
68 if (acquired) {
69 __atomic_store_n(&holder_, __builtin_return_address(0),
70 __ATOMIC_RELAXED);
71 }
72 return acquired;
73 }
74
75 const void *holder() const noexcept {
76 return __atomic_load_n(&holder_, __ATOMIC_RELAXED);
77 }
78
79 private:
80 int locked_ = 0;
81 const void *holder_ = nullptr;
82};
