77inline uint64_t mb2_find_tag(uint32_t type) {
78 if (multiboot_magic != 0x36D76289) return 0;
79
80 auto* info = reinterpret_cast<Multiboot2Info*>(
81 static_cast<uint64_t>(multiboot_info_ptr));
82
83 uint64_t addr = multiboot_info_ptr + 8;
84 while (addr < multiboot_info_ptr + info->total_size) {
85 auto* tag = reinterpret_cast<Multiboot2Tag*>(addr);
86 if (tag->type == 0) break;
87 if (tag->type == type) return addr;
88 addr += (tag->size + 7) & ~7;
89 }
90 return 0;
91}
