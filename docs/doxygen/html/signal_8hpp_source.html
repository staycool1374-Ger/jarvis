73static inline ExceptionSignalMap exception_to_signal(uint64_t vector) {
74 switch (vector) {
75 case 0: return {Signal::SIGFPE, SignalAction::TERMINATE, "SIGFPE"}; // #DE
76 case 4: return {Signal::SIGILL, SignalAction::TERMINATE, "SIGILL"}; // #OF
77 case 5: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #BR
78 case 6: return {Signal::SIGILL, SignalAction::TERMINATE, "SIGILL"}; // #UD
79 case 7: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #NM
80 case 8: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #DF
81 case 10: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #TS
82 case 11: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #NP
83 case 12: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #SS
84 case 13: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #GP
85 case 14: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #PF
86 case 16: return {Signal::SIGFPE, SignalAction::TERMINATE, "SIGFPE"}; // #MF
87 case 17: return {Signal::SIGSEGV, SignalAction::TERMINATE, "SIGSEGV"}; // #AC
88 case 19: return {Signal::SIGFPE, SignalAction::TERMINATE, "SIGFPE"}; // #XF
89 default: return {Signal::SIGSYS, SignalAction::TERMINATE, "SIGSYS"};
90 }

123static inline SignalAction default_signal_action(uint64_t sig) {
124 switch (static_cast<Signal>(sig)) {
125 case Signal::SIGCHLD:
126 case Signal::SIGCONT:
127 return SignalAction::IGNORE;
128 case Signal::SIGSTOP:
129 case Signal::SIGTSTP:
130 return SignalAction::STOP;
131 default:
132 return SignalAction::TERMINATE;
133 }
