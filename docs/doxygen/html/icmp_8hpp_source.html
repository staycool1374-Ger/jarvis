38inline uint16_t icmp_checksum(const uint8_t* data, size_t len) {
39 uint32_t sum = 0;
40 auto* words = reinterpret_cast<const uint16_t*>(data);
41 size_t word_count = len / 2;
42 for (size_t i = 0; i < word_count; ++i) {
43 sum += words[i];
44 }
45 if (len & 1) {
46 sum += static_cast<uint16_t>(data[len - 1]) << 8;
47 }
48 while (sum >> 16) sum = (sum & 0xFFFF) + (sum >> 16);
49 return static_cast<uint16_t>(~sum & 0xFFFF);
50}
