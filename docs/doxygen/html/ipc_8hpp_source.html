41struct MessageQueue {
42 Message msgs[IPC_MAX_QUEUE_MSG];
43 uint64_t prio_bitmap;
44 // NOTE: head/tail/count are accessed from a task that busy-waits on
45 // is_empty()/is_full() across reschedule() calls. reschedule() only
46 // *defers* the context switch (the peer task's push/pop happens via an ISR
47 // context-switch, invisible to the compiler's data-flow analysis for the
48 // current function), so a plain field would be hoisted/cached by the
49 // optimizer and the waiter would never observe the peer's write — wedging
50 // IPC send_sync (ipc_blocking test hang). std::atomic (relaxed is enough
51 // on the single kernel stack / UP) forces a fresh load on every access.
52 volatile size_t head;
53 volatile size_t tail;
54 volatile size_t count;
55
56 TaskControlBlock *blocked_senders_head;
57 TaskControlBlock *blocked_senders_tail;
58
60 TaskControlBlock *owner;
61
62 MessageQueue()
63 : prio_bitmap(0), head(0), tail(0), count(0),
64 blocked_senders_head(nullptr), blocked_senders_tail(nullptr),
65 owner(nullptr) {
66 }
67
69 ~MessageQueue();
70
71 sync::SpinLock lock_;
72
74 void init();
77 bool push(const Message &msg);
80 bool pop(Message &msg);
81
82 bool is_empty() const {
83 return count == 0;
84 }
85 bool is_full() const {
86 return count >= IPC_MAX_QUEUE_MSG;
87 }
88
91 size_t highest_priority() const;
92};
