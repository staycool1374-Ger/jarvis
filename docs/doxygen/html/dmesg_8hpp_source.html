41template<size_t Capacity>
42class DmesgBuffer {
43 static_assert((Capacity & (Capacity - 1)) == 0, "Capacity must be power of 2");
44 static constexpr size_t MASK = Capacity - 1;
45
46 LogEntry buffer[Capacity];
47 alignas(64) volatile size_t head{0};
48 alignas(64) volatile size_t tail{0};
49
50public:
51 bool push(ErrorSubsystem subsys, uint64_t err_code, const char* msg, uintptr_t ctx = 0);
52 bool pop(LogEntry& entry);
53
54 template<typename Fn>
55 void for_each(Fn&& fn) const {
56 size_t t = atomic_load(&tail, __ATOMIC_ACQUIRE);
57 size_t h = atomic_load(&head, __ATOMIC_ACQUIRE);
58
59 while (t != h) {
60 fn(buffer[t]);
61 t = (t + 1) & MASK;
62 }
63 }
64
65 bool empty() const;
66 size_t size() const;
67 void clear();
68
69 size_t head_index() const { return head; }
70 size_t tail_index() const { return tail; }
71};
