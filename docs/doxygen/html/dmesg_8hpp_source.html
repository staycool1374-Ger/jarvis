85 template <typename Fn> void for_each(Fn &&fn) const {
86 size_t t = atomic_load(&tail, __ATOMIC_ACQUIRE);
87 size_t h = atomic_load(&head, __ATOMIC_ACQUIRE);
88
89 while (t != h) {
90 fn(buffer[t]);
91 t = (t + 1) & MASK;
92 }
93 }

148inline const char *base_error_string(uint64_t code) {
149 // Custom event ranges
150 if ((code & ~0xFFULL) == 0xDA00) {
151 switch (code) {
152 case 0xDA01:
153 return "Daemon exited";
154 case 0xDA02:
155 return "Daemon restarted";
156 case 0xDA03:
157 return "Daemon ensured";
158 case 0xDA04:
159 return "Daemon terminated";
160 case 0xDA05:
161 return "Daemon restarting";
162 case 0xDA06:
163 return "Daemon up";
164 }
165 return "Daemon event";
166 }
167 // Standard kernel::Error range (0–9)
168 switch (static_cast<kernel::Error>(code)) {
169 case kernel::Error::OK:
170 return "OK";
171 case kernel::Error::OOM:
172 return "Out of memory";
173 case kernel::Error::INVALID_ARG:
174 return "Invalid argument";
175 case kernel::Error::NOT_FOUND:
176 return "Not found";
177 case kernel::Error::ALREADY_EXISTS:
178 return "Already exists";
179 case kernel::Error::TIMEOUT:
180 return "Timeout";
181 case kernel::Error::BUSY:
182 return "Busy";
183 case kernel::Error::NOT_IMPLEMENTED:
184 return "Not implemented";
185 case kernel::Error::IO_ERROR:
186 return "I/O error";
187 case kernel::Error::CORRUPTED:
188 return "Corrupted";
189 }
190 return "Unknown base error";
191}

216inline const char *error_string(ErrorSubsystem subsys, uint64_t code) {
217 switch (subsys) {
218 case ErrorSubsystem::BASE:
219 return base_error_string(code);
220 case ErrorSubsystem::SYNC:
221 return kernel::errors::error_string(
222 static_cast<kernel::errors::SyncError>(code));
223 case ErrorSubsystem::VFS:
224 return kernel::errors::error_string(
225 static_cast<kernel::errors::VfsError>(code));
226 case ErrorSubsystem::MEMPOOL:
227 return kernel::errors::error_string(
228 static_cast<kernel::errors::MemPoolError>(code));
229 case ErrorSubsystem::SCHED:
230 return kernel::errors::error_string(
231 static_cast<kernel::errors::SchedulerError>(code));
232 case ErrorSubsystem::IPC:
233 return kernel::errors::error_string(
234 static_cast<kernel::errors::IpcError>(code));
235 case ErrorSubsystem::SYSCALL:
236 return kernel::errors::error_string(
237 static_cast<kernel::errors::SyscallError>(code));
238 default:
239 return "UNKNOWN";
240 }
