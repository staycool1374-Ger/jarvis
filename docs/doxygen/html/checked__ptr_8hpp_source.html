36static inline bool is_user_range(const void* user_ptr, uint64_t size) {
37 if (!user_ptr) return false;
38 uint64_t addr = reinterpret_cast<uint64_t>(user_ptr);
39 if (addr >= USER_SPACE_LIMIT) return false;
40 if (size == 0) return true;
41 uint64_t end = addr + size - 1;
42 if (end < addr) return false;
43 if (end >= USER_SPACE_LIMIT) return false;
44 return true;
45}

47template <kernel::TriviallyCopiable T>
48class CheckedPtr {
49 uint64_t addr_;
50 uint64_t count_;
51public:
52 CheckedPtr() : addr_(0), count_(0) {}

54 CheckedPtr(T* user_ptr, uint64_t count = 1)
55 : addr_(reinterpret_cast<uint64_t>(user_ptr)), count_(count) {}

59 bool valid() const {
60 if (count_ == 0) return true;
61 if (addr_ == 0) return false;
62 return is_user_range(reinterpret_cast<const void*>(addr_),
63 count_ * sizeof(T));
64 }

113static inline bool is_user_string(const void* user_ptr, uint64_t max_len = 4096
114 ) {
115 uint64_t addr = reinterpret_cast<uint64_t>(user_ptr);
116 if (addr >= USER_SPACE_LIMIT || addr == 0) return false;
117 uint64_t end = addr + max_len - 1;
118 if (end < addr || end >= USER_SPACE_LIMIT) return false;
119 auto* p = static_cast<const volatile char*>(user_ptr);
120 for (uint64_t i = 0; i < max_len; ++i) {
121 if (p[i] == '\0') return true;
122 }
123 return false;
124}

128static inline bool strncpy_from_user(char* dst, const char* src,
129 uint64_t max_len) {
130 if (!is_user_string(src, max_len)) return false;
131 for (uint64_t i = 0; i < max_len; ++i) {
132 dst[i] = src[i];
133 if (src[i] == '\0') return true;
134 }
135 dst[max_len - 1] = '\0';
136 return true;

143static inline bool safe_copy_from_user(T* dst, const T* src, uint64_t count) {
144 if (!is_user_range(src, count * sizeof(T))) return false;
