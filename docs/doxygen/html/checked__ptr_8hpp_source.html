39static inline bool is_user_range(const void *user_ptr, uint64_t size) {
40 if (!user_ptr)
41 return false;
42 uint64_t addr = reinterpret_cast<uint64_t>(user_ptr);
43 if (addr >= USER_SPACE_LIMIT)
44 return false;
45 if (size == 0)
46 return true;
47 uint64_t end = addr + size - 1;
48 if (end < addr)
49 return false;
50 if (end >= USER_SPACE_LIMIT)
51 return false;
52 return true;
53}

59template <kernel::TriviallyCopiable T> class CheckedPtr {

63 public:

71 CheckedPtr(T *user_ptr, uint64_t count = 1)
72 : addr_(reinterpret_cast<uint64_t>(user_ptr)), count_(count) {

77 bool valid() const {
78 if (count_ == 0)
79 return true;
80 if (addr_ == 0)
81 return false;
82 return is_user_range(reinterpret_cast<const void *>(addr_),
83 count_ * sizeof(T));
84 }

140static inline bool is_user_string(const void *user_ptr,
141 uint64_t max_len = 4096) {
142 uint64_t addr = reinterpret_cast<uint64_t>(user_ptr);
143 if (addr >= USER_SPACE_LIMIT || addr == 0)
144 return false;
145 uint64_t end = addr + max_len - 1;
146 if (end < addr || end >= USER_SPACE_LIMIT)
147 return false;
148 auto *p = static_cast<const volatile char *>(user_ptr);
149 for (uint64_t i = 0; i < max_len; ++i) {
150 if (p[i] == '\0')
151 return true;
152 }
153 return false;
154}

158static inline bool strncpy_from_user(char *dst, const char *src,
