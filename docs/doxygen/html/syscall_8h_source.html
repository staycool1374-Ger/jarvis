78static inline long __syscall5(long num, long a0, long a1, long a2, long a3) {
79 long ret;
80#if defined(__x86_64__)
81 asm volatile(
82 "int $0x80"
83 : "=a"(ret)
84 : "a"(num), "b"(a0), "c"(a1), "d"(a2), "S"(a3)
85 : "memory", "cc"
86 );
87#elif defined(__aarch64__)
88 register long x8 asm("x8") = num;
89 register long x0 asm("x0") = a0;
90 register long x1 asm("x1") = a1;
91 register long x2 asm("x2") = a2;
92 register long x3 asm("x3") = a3;
93 register long x4 asm("x4") = 0;
94 asm volatile(
95 "svc #0"
96 : "+r"(x0)
97 : "r"(x8), "r"(x1), "r"(x2), "r"(x3), "r"(x4)
98 : "memory"
99 );
100 ret = x0;
101#elif defined(__riscv) && __riscv_xlen == 64
102 register long a7 asm("a7") = num;
103 register long a0_reg asm("a0") = a0;
104 register long a1_reg asm("a1") = a1;
105 register long a2_reg asm("a2") = a2;
106 register long a3_reg asm("a3") = a3;
107 asm volatile(
108 "ecall"
109 : "+r"(a0_reg)
110 : "r"(a7), "r"(a1_reg), "r"(a2_reg), "r"(a3_reg)
111 : "memory"
112 );
113 ret = a0_reg;
114#else
115# error "Unsupported architecture for __syscall5"
116#endif
117 return ret;
118}
