164 struct TaskIter {
165 uint64_t idx;
166 explicit TaskIter(uint64_t start = 0) : idx(start) {}
167 TaskControlBlock* next(TaskControlBlock* exclude = nullptr) {
168 while (idx < task_count_) {
169 auto* t = tasks_[idx++];
170 if (t && t->magic == TaskControlBlock::TCB_MAGIC && t != exclude)
171 return t;
172 }
173 return nullptr;
174 }
175 };
