307 struct TaskIter {
308 uint64_t idx;
309 TaskControlBlock *cur_;
310 explicit TaskIter(uint64_t start = 0)
311 : idx(start), cur_(Scheduler::all_tasks_.first_ptr()) {
312 for (uint64_t i = 0; i < start; ++i) {
313 cur_ = Scheduler::all_tasks_.next_ptr(cur_);
314 if (!cur_)
315 break;
316 }
317 }
318 TaskControlBlock *next(TaskControlBlock *exclude = nullptr) {
319 while (cur_) {
320 auto *t = cur_;
321 cur_ = Scheduler::all_tasks_.next_ptr(cur_);
322 ++idx;
323 if (t && t->magic == TaskControlBlock::TCB_MAGIC &&
324 t != exclude)
325 return t;
326 }
327 return nullptr;
328 }
329 };
