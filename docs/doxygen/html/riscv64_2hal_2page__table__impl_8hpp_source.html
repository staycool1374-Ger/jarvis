12// Sv39: 3-level page table (512 entries each)
13// L0 -> 1GB regions (bits 38:30)
14// L1 -> 2MB regions (bits 29:21)
15// L2 -> 4KB pages (bits 20:12)

17inline uint64_t arch_page_table_current() {
18 uint64_t satp;
19 asm volatile("csrr %0, satp" : "=r"(satp) : : "memory");
20 return (satp & 0xFFFFFFFFFFF) << 12; // PPN -> physical address
21}

23inline void arch_page_table_activate(uint64_t phys_ppn_shifted) {

78inline uint64_t* ArchPageTable::get_table(uint64_t table_base, uint64_t index, bool create) {
79 uint64_t* table = reinterpret_cast<uint64_t*>(table_base);
80 uint64_t entry = table[index];
81
82 if (entry & DESC_VALID) {
83 return reinterpret_cast<uint64_t*>(entry & ~0xFFF);
84 }
85
86 if (!create) return nullptr;
87
88 uint64_t new_page = kernel::PMM::alloc_page_table();
89 if (!new_page) return nullptr;
90
91 void* new_page_ptr = reinterpret_cast<void*>(new_page);
92 memset(new_page_ptr, 0, PAGE_SIZE);
93 // Mark as valid, readable, writable table pointer
94 table[index] = (new_page & ~0xFFF) | DESC_VALID | DESC_TABLE | DESC_R | DESC_W;
95 return reinterpret_cast<uint64_t*>(new_page);
96}

98inline uint64_t ArchPageTable::attr_from_flags(uint64_t flags) {
99 uint64_t attr = DESC_VALID | DESC_AF;
100 (void)DESC_G;
101 if (flags & kernel::PageFlags::PRESENT) attr |= DESC_VALID;
102 if (!(flags & kernel::PageFlags::NX)) attr |= DESC_X;
103 if (flags & kernel::PageFlags::WRITE) attr |= DESC_W;
104 if (flags & kernel::PageFlags::USER) attr |= DESC_USER;
105 // Default to readable unless explicitly no-access
106 if (flags & kernel::PageFlags::PRESENT) attr |= DESC_R;
107 return attr;
108}

122inline kernel::ErrorOr<void> ArchPageTable::map_page(uint64_t virt, uint64_t phys, uint64_t flags) {
123 uint64_t l0_idx = (virt >> L0_SHIFT) & TABLE_MASK;
124 uint64_t l1_idx = (virt >> L1_SHIFT) & TABLE_MASK;
125 uint64_t l2_idx = (virt >> L2_SHIFT) & TABLE_MASK;
126
127 uint64_t l0_phys = current();
128 uint64_t* l1 = walk_l0(l0_phys, l0_idx, true);
129 if (!l1) return kernel::Error::OOM;
130
131 uint64_t* l2 = walk_l1(reinterpret_cast<uint64_t>(l1), l1_idx, true);
132 if (!l2) return kernel::Error::OOM;
133
134 uint64_t entry_flags = attr_from_flags(flags);
135
136 // Try 2MB block mapping if aligned
137 if ((virt & ((1ULL << L1_SHIFT) - 1)) == 0 && (phys & ((1ULL << L1_SHIFT) - 1)) == 0) {
138 l2[l1_idx] = (phys & ~0x1FFFFF) | entry_flags | DESC_BLOCK | (1ULL << 6);
139 return {};
140 }
141
142 // 4KB page at L2
143 l2[l2_idx] = (phys & ~0xFFF) | entry_flags | DESC_PAGE | (1ULL << 6) | (1ULL << 7);
144 return {};
145}

147inline kernel::ErrorOr<void> ArchPageTable::unmap_page(uint64_t virt) {
148 uint64_t l0_idx = (virt >> L0_SHIFT) & TABLE_MASK;
149 uint64_t l1_idx = (virt >> L1_SHIFT) & TABLE_MASK;
150 uint64_t l2_idx = (virt >> L2_SHIFT) & TABLE_MASK;
151
152 uint64_t l0_phys = current();
153 uint64_t* l0 = reinterpret_cast<uint64_t*>(l0_phys);
154
155 if (!(l0[l0_idx] & DESC_VALID)) return kernel::Error::NOT_FOUND;
156 uint64_t* l1 = reinterpret_cast<uint64_t*>(l0[l0_idx] & ~0xFFF);
157 if (!(l1[l1_idx] & DESC_VALID)) return kernel::Error::NOT_FOUND;
158 uint64_t* l2 = reinterpret_cast<uint64_t*>(l1[l1_idx] & ~0xFFF);
159
160 // Check for 2MB block
161 if ((virt & ((1ULL << L1_SHIFT) - 1)) == 0 && !(l2[l1_idx] & DESC_TABLE)) {
162 l2[l1_idx] = 0;
163 return {};
164 }
165
166 if (!(l2[l2_idx] & DESC_VALID)) return kernel::Error::NOT_FOUND;
167 l2[l2_idx] = 0;
168 return {};
169}

171inline uint64_t ArchPageTable::get_physical(uint64_t virt) {
172 uint64_t l0_idx = (virt >> L0_SHIFT) & TABLE_MASK;
173 uint64_t l1_idx = (virt >> L1_SHIFT) & TABLE_MASK;
174 uint64_t l2_idx = (virt >> L2_SHIFT) & TABLE_MASK;
175
176 uint64_t l0_phys = current();
177 uint64_t* l0 = reinterpret_cast<uint64_t*>(l0_phys);
178
179 if (!(l0[l0_idx] & DESC_VALID)) return 0;
180 uint64_t* l1 = reinterpret_cast<uint64_t*>(l0[l0_idx] & ~0xFFF);
181 if (!(l1[l1_idx] & DESC_VALID)) return 0;
182 uint64_t* l2 = reinterpret_cast<uint64_t*>(l1[l1_idx] & ~0xFFF);
183
184 // 2MB block mapping
185 if ((virt & ((1ULL << L1_SHIFT) - 1)) == 0 && !(l2[l1_idx] & DESC_TABLE)) {
186 return (l2[l1_idx] & ~0x1FFFFF) | (virt & 0x1FFFFF);
187 }
188
189 if (!(l2[l2_idx] & DESC_VALID)) return 0;
190 return (l2[l2_idx] & ~0xFFF) | (virt & 0xFFF);
191}
