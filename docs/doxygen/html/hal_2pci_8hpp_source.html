47// ---------------------------------------------------------------------------
48// Architecture-specific PCI config space access
49// x86_64: CF8/CFC port I/O (0xCF8 / 0xCFC)
50// aarch64 / riscv64: ECAM memory-mapped (QEMU virt at 0x100000000)
51//
52// All access functions work on dword-aligned addresses internally, matching
53// the x86 CF8/CFC semantics where bits [1:0] of the register offset select
54// the byte/word within the dword. This ensures identical behavior across
55// all architectures.
56// ---------------------------------------------------------------------------

75inline uint64_t pci_make_addr(PciBdf bdf, uint8_t reg) {
76 return 0x80000000U | (static_cast<uint32_t>(bdf.bus) << 16) |
77 (static_cast<uint32_t>(bdf.device) << 11) |
78 (static_cast<uint32_t>(bdf.function) << 8) | reg;
79}

84inline uint32_t pci_config_readl(uint64_t addr) {
85 outl(PCI_CONFIG_ADDR, static_cast<uint32_t>(addr & ~3ULL));
86 return inl(PCI_CONFIG_DATA);
87}
88
92inline uint8_t pci_config_readb(uint64_t addr) {
93 uint32_t val = pci_config_readl(addr);
94 return static_cast<uint8_t>((val >> ((addr & 3) * 8)) & 0xFF);
95}
96
100inline uint16_t pci_config_readw(uint64_t addr) {
101 uint32_t val = pci_config_readl(addr);
102 return static_cast<uint16_t>((val >> ((addr & 2) * 8)) & 0xFFFF);
103}
104
108inline void pci_config_writel(uint64_t addr, uint32_t val) {
109 outl(PCI_CONFIG_ADDR, static_cast<uint32_t>(addr & ~3ULL));
110 outl(PCI_CONFIG_DATA, val);
111}
112
117inline void pci_config_writew(uint64_t addr, uint16_t val) {
118 uint32_t old = pci_config_readl(addr);
119 uint32_t shift = (addr & 2) * 8;
120 uint32_t new_val =
121 (old & ~(0xFFFF << shift)) | (static_cast<uint32_t>(val) << shift);
122 pci_config_writel(addr, new_val);
123}
